Commands for VLAN Configuration:-

Switch 1:

-- to configure VLAN 10

enable
configure terminal
vlan 10
name HR
exit

-- to configure VLAN 20

vlan 20
name Sales
exit

-- to assign access ports to the respective VLAN's

interface range fastEthernet 0/1 -2
switchport access vlan 10
exit

interface range fastEthernet 0/3 -4
switchport access vlan 20
exit
exit

-- to view VLAN's

show vlan brief
